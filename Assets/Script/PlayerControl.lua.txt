function update()
    moneyAmout = CS.UnityEngine.PlayerPrefs.GetInt("MoneyAmout");
    moneyAmoutTextComp.text = moneyAmout;

    if CS.UnityEngine.Input.GetMouseButtonDown(0) and isJumping == false then
        rb.velocity = CS.UnityEngine.Vector2(0, 7);
        isJumping = false;
        fly:Play();
    end
    
end

function onCollisionEnter2D(collision)
    isJumping = false;
end

function onTriggerEnter2D(collision)
            
    if collision.tag == "GameOver" then
        CS.UnityEngine.Time.timeScale = 0;
        gameManagerComp:GameOver();
    end
    
    if collision.tag == "StarClam" then
        moneyAmout = moneyAmout + 1;
        CS.UnityEngine.PlayerPrefs.SetInt("MoneyAmout", moneyAmout);
        clam:Play();
    end
end
