local score = 0;
local timer = 0;
local maxTime = 0;

function start()
    score = 0;
    maxTime = 0.15;
end

function update()
    timer = timer + CS.UnityEngine.Time.deltaTime;
    if timer >= maxTime then
        score = score +1;
        timer = 0;
        if score % 100 == 0 then
            CS.UnityEngine.Time.timeScale = CS.UnityEngine.Time.timeScale + 0.05; 
        end
    end
end
